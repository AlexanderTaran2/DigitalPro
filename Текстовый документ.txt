    <script>
        let screen = document.getElementById('screen');
        let value = '0';

        function add(x) {
            if (value === 'Error' || value === '0' && x !== ',') {
                value = x;
            } else {
                value += x;
            }

            screen.textContent = value;
        }

        function back() {
            if (value.length > 1) {
                value = value.slice(0, -1);
            } else {
                value = '0';
            }

            screen.textContent = value;
        }

        function clearAll() {
            value = '0';
            screen.textContent = value;
        }

        function calc() {
            try {
                value = eval(value).toString();
            } catch {
                value = 'Error'
                setTimeout(() => {
                    if (value === 'Error') clearAll();
                }, 2500);
            }
            screen.textContent = value;
        }
    </script>
